const { Sequelize, DataTypes } = require('sequelize'); const sequelize = new Sequelize(process.env.POSTGRES_URI); const User = sequelize.define('User', { fullName: { type: DataTypes.STRING, allowNull: false }, dob: { type: DataTypes.DATEONLY, allowNull: false }, nationalId: { type: DataTypes.STRING, allowNull: false, unique: true }, email: { type: DataTypes.STRING, allowNull: true } }, { timestamps: true }); module.exports = User;
