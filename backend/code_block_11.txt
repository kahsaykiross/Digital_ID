const { Sequelize } = require('sequelize'); const mongoose = require('mongoose'); const sequelize = new Sequelize(process.env.POSTGRES_URI, { dialect: 'postgres' }); mongoose.connect(process.env.MONGO_URI, { useNewUrlParser: true, useUnifiedTopology: true }).then(() => console.log('MongoDB connected')) .catch(err => console.error('MongoDB connection error:', err)); module.exports = { sequelize, mongoose };
