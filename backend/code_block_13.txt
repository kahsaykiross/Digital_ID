const { DataTypes } = require('sequelize'); const { sequelize } = require('../config/db'); const User = sequelize.define('User', { fullName: { type: DataTypes.STRING, allowNull: false }, dob: { type: DataTypes.DATEONLY, allowNull: false }, nationalId: { type: DataTypes.STRING, allowNull: false, unique: true }, email: { type: DataTypes.STRING, allowNull: true }, password: { type: DataTypes.STRING, allowNull: false } }, { timestamps: true }); module.exports = User;
