import React, { useState } from 'react'; import { BrowserRouter as Router, Routes, Route } from 'react-router-dom'; import Login from './components/Login'; import Dashboard from './components/Dashboard'; function App() { const [token, setToken] = useState(localStorage.getItem('adminToken') || null); return ( <Router> <Routes> <Route path="/" element={token ? <Dashboard token={token} /> : <Login setToken={setToken} />} /> </Routes> </Router> ); } export default App;
