import axios from 'axios'; const API_URL = process.env.REACT_APP_API_URL || 'http://localhost:5000/api'; export const adminLogin = (data) => axios.post(`${API_URL}/admin/login`, data); export const getAllUsers = (token) => axios.get(`${API_URL}/admin/users`, { headers: { Authorization: `Bearer ${token}` } }); export const getUserDetails = (userId, token) => axios.get(`${API_URL}/admin/users/${userId}`, { headers: { Authorization: `Bearer ${token}` } }); export const approveUser = (userId, token) => axios.post(`${API_URL}/admin/users/${userId}/approve`, {}, { headers: { Authorization: `Bearer ${token}` } });
